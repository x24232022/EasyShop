} //失败 { "code": 2, "msg": "failed" }

商品模块

1.获取所有商品

路径： GoodsServlet?method=getAll

方法： post

请求： 

//商品每页20条数据
pageNo=1  //string
//类型
type=""  //全部商品传空字符串即可

类型共有household、electron、dress、toy、book、gift、other 分别对应家用、电子、服饰、玩具、图书、礼品、其它"

响应体：

//成功
{
    "code": 1,
    "msg": " success",
    "datas": [
            {
                "price": "66",        //商品价格
                "name": "单车",      //商品名称
                "description": "......",  //商品描述
                "page": "/images/D3228118230A43C0B77/5606FF8A48F1FC4907D/F99E38F09A.JPEG", //商品的第一张图片
                "type": "other",      //商品类型
                "uuid": "5606FF8EF60146A48F1FCDC34144907D",  //商品表中主键
                "master": "android"  //发布者
            },
            {
               ......
            },
            .....
        ]
}

//失败
{
    "code": 2,
    "msg": "failed"
}

3.添加商品

路径： GoodsServlet?method=add

方法： post/Multipart

请求：

//请求是一个商品类的JSON字符串， 商品的图片以文件形式上传

{
    "description": "诚信商家，非诚勿扰",     //商品描述
    "master": "android",                    //商品发布者
    "name": "礼物，鱼丸，鱼翅，火箭，飞机",   //商品名称
    "price": "88",                          //商品价格
    "type": "gift"                          //商品类型
}

响应体：

//成功
{
    "code": 1,
    "msg": "添加成功！"
}
